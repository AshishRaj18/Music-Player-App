1    package com.ashish.mymusic
2    
3    import android.content.Intent
4    import android.os.Bundle
5    import android.os.Handler
6    import android.view.WindowManager
7    import androidx.appcompat.app.AppCompatActivity
8    
9    @Suppress("DEPRECATION")
10   
11   class SplashActivity : AppCompatActivity() {
12       override fun onCreate(savedInstanceState: Bundle?) {
13           super.onCreate(savedInstanceState)
14           setContentView(R.layout.activity_splash)
15   
16   
17           // Take instance of Action Bar
18           // using getSupportActionBar and
19           // if it is not Null
20           // then call hide function
21           if (getSupportActionBar() != null) {
22               getSupportActionBar()?.hide();
23           }
24   
25           // This is used to hide the status bar and make
26           // the splash screen as a full screen activity.
27           window.setFlags(
28               WindowManager.LayoutParams.FLAG_FULLSCREEN,
29               WindowManager.LayoutParams.FLAG_FULLSCREEN
30           )
31   
32           // we used the postDelayed(Runnable, time) method
33           // to send a message with a delayed time.
34           Handler().postDelayed({
35               val intent = Intent(this, MainActivity::class.java)
36               startActivity(intent)
37               finish()
38           }, 2000) // 3000 is the delayed time in milliseconds.
39       }
40   }
