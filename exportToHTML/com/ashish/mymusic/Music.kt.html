9    fun formatDuration(duration: Long):String{
10       val minutes=TimeUnit.MINUTES.convert(duration,TimeUnit.MILLISECONDS)
11       val seconds=(TimeUnit.SECONDS.convert(duration,TimeUnit.MILLISECONDS) -
12               minutes*TimeUnit.SECONDS.convert(1,TimeUnit.MINUTES))
13   
14       return String.format("%02d:%02d",minutes,seconds)
